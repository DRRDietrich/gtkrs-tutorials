Creating a Window with a Button

First, let's make ourselves a convenience function for blocking on a future:


#![allow(unused)]
fn main() {
use std::future::Future;

fn block_on<F>(future: F) -> F::Output where F: Future {
    glib::MainContext::default().block_on(future)
}
}

Then, we'll create an App struct that will hold our most important widgets and state that we'll interact with throughout application lifetime. To start out, this will be a simple gtk::Container that contains a single gtk::Button. The button will have a border of 4 units around it, and upon clicking it, it will send an Event::Clicked. Once we create the gtk::Button, we'll create a gtk::Box and add the button to it. It's important to note that you have to call .show_all() on the container to have the widgets inside of the container shown in the app.


#![allow(unused)]
fn main() {
#[macro_use]
extern crate cascade;

use async_channel::Sender;

struct App {
    pub container: gtk::Box,
    pub button: gtk::Button,
    pub clicked: u32,
}

impl App {
    pub fn new(tx: Sender<Event>) -> Self {
        let button = cascade! {
            gtk::Button::with_label("Click Me");
            ..set_border_width(4);
            ..connect_clicked(move |_| {
                block_on(tx.send(Event::Clicked));
            });
        };

        let container = cascade! {
            gtk::Box::new(gtk::Orientation::Vertical, 0);
            ..add(&button);
            ..show_all();
        };

        Self { container, button, clicked: 0 }
    }
}
}

Then we'll add this to our application and attach it to a "Toplevel" gtk::Window:


#![allow(unused)]
fn main() {
    let (tx, rx) = async_channel::unbounded();

    let mut app = App::new(tx);

    cascade! {
        gtk::Window::new(gtk::WindowType::Toplevel);
        ..set_title("First GTK App");
        ..add(&app.container);
        ..connect_delete_event(move |_, _| {
            gtk::main_quit();
            gtk::Inhibit(false)
        });
        ..show_all();
    };

    gtk::Window::set_default_icon_name("icon-name-here");

    // Processes all application events received from signals
    let event_handler = async move {
        while let Ok(event) = rx.recv().await {
            match event {
                Event::Clicked => {
                    app.clicked += 1;
                    app.button.set_label(&format!("I have been clicked {} times", app.clicked));
                }
            }
        }
    };

    // GLib has an executor in the background that will
    // asynchronously handle our events on this thread
    glib::MainContext::default().spawn_local(event_handler);
}

Note that we have programmed the delete_event to call gtk::main_quit() when it is invoked. This is called when you exit the program by keyboard shortcut or clicking the close button. You can also set an icon to be shown for the application by invoking gtk::Window::set_default_icon_name("icon-name-here");. In our event-handler, we increment the app.clicked state each time it's called, and then update the button's label.

Your application source code should look like this now:

#[macro_use]
extern crate cascade;

use async_channel::Sender;
use gtk::prelude::*;
use std::future::Future;
use std::process;

fn block_on<F>(future: F) -> F::Output where F: Future {
    glib::MainContext::default().block_on(future)
}

enum Event {
    Clicked
}

struct App {
    pub container: gtk::Box,
    pub button: gtk::Button,
    pub clicked: u32,
}

impl App {
    pub fn new(tx: Sender<Event>) -> Self {
        let button = cascade! {
            gtk::Button::with_label("Click Me");
            ..set_border_width(4);
            ..connect_clicked(move |_| {
                let _ = block_on(tx.send(Event::Clicked));
            });
        };

        let container = cascade! {
            gtk::Box::new(gtk::Orientation::Vertical, 0);
            ..add(&button);
            ..show_all();
        };

        Self { container, button, clicked: 0 }
    }
}

fn main() {
    glib::set_program_name("First GTK App".into());
    glib::set_application_name("First GTK App");

    // Initialize GTK before proceeding.
    if gtk::init().is_err() {
        eprintln!("failed to initialize GTK Application");
        process::exit(1);
    }

    // Attach `tx` to our widgets, and `rx` to our event handler
    let (tx, rx) = async_channel::unbounded();

    let mut app = App::new(tx);

    cascade! {
        gtk::Window::new(gtk::WindowType::Toplevel);
        ..set_title("First GTK App");
        ..add(&app.container);
        ..connect_delete_event(move |_, _| {
            gtk::main_quit();
            gtk::Inhibit(false)
        });
        ..show_all();
    };

    gtk::Window::set_default_icon_name("icon-name-here");

    // Processes all application events received from signals
    let event_handler = async move {
        while let Ok(event) = rx.recv().await {
            match event {
                Event::Clicked => {
                    app.clicked += 1;
                    app.button.set_label(&format!("I have been clicked {} times", app.clicked));
                }
            }
        }
    };

    // GLib has an executor in the background that will
    // asynchronously handle our events on this thread
    glib::MainContext::default().spawn_local(event_handler);

    // Thread will block here until the application is quit
    gtk::main();
}
